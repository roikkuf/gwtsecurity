001    package com.gwt.ss;
002    
003    import java.lang.reflect.Field;
004    import javax.servlet.FilterChain;
005    import javax.servlet.ServletContext;
006    import org.aspectj.lang.JoinPoint;
007    import org.aspectj.lang.ProceedingJoinPoint;
008    import org.aspectj.lang.annotation.Around;
009    import org.aspectj.lang.annotation.Aspect;
010    import org.slf4j.Logger;
011    import org.slf4j.LoggerFactory;
012    import org.springframework.security.core.Authentication;
013    import org.springframework.security.core.context.SecurityContextHolder;
014    import org.springframework.security.core.session.SessionInformation;
015    import org.springframework.security.core.session.SessionRegistry;
016    import org.springframework.security.web.authentication.logout.LogoutHandler;
017    import org.springframework.security.web.authentication.session.SessionAuthenticationException;
018    import org.springframework.security.web.authentication.session.SessionAuthenticationStrategy;
019    import org.springframework.security.web.authentication.session.SessionFixationProtectionStrategy;
020    import org.springframework.security.web.session.SessionManagementFilter;
021    import org.springframework.security.web.context.SecurityContextRepository;
022    import org.springframework.security.web.session.ConcurrentSessionFilter;
023    import org.springframework.web.context.ServletContextAware;

025    /**
026     * If you apply <a href="http://static.springsource.org/spring-security/site/docs/3.1.x/reference/ns-config.html#ns-session-mgmt">
027     * Session Management</a> in your secured gwt application,you need to make sure
028     * <ul>
029     *   <li>property &quot;proxy-target-class&quot; of &lt;aop:aspectj-autoproxy&gt; set as &quot;false&quot; owing to 
030     *   Spring's <a href="http://static.springsource.org/spring/docs/3.1.0.M1/spring-framework-reference/html/aop.html#aop-proxying">proxy
031     *   mechanism</a> only allow null constructor,unfortunately {@link org.springframework.security.web.session.SessionManagementFilter SessionManagementFilter}
032     *   didn't. 
033     *   </li> 
034     *   <li>Instantiate this bean</li>
035     * </ul></br>
036     * 如果您的GWT程式設定了<a href="http://static.springsource.org/spring-security/site/docs/3.1.x/reference/ns-config.html#ns-session-mgmt">
037     * Session Management</a>，您必須
038     * <ul>
039     *   <li>將&lt;aop:aspectj-autoproxy&gt;的屬性&quot;proxy-target-class&quot;設為&quot;false&quot;，這是由於
040     *   Sprig AOP的<a href="http://static.springsource.org/spring/docs/3.1.0.M1/spring-framework-reference/html/aop.html#aop-proxying">proxy
041     *   機制</a>不接受有參數的建構子，而{@link org.springframework.security.web.session.SessionManagementFilter SessionManagementFilter}
042     *   的建構子卻含有參數        
043     *   </li>  
044     *   <li>建立本類別實例</li> 
045     * </ul>       
046     */
047    @Aspect
048    public class GwtSessionManagement implements ServletContextAware {
049    
050        protected static Logger logger = LoggerFactory.getLogger(GwtSessionManagement.class);
051        private ServletContext servletContext;
052        static final String FILTER_APPLIED = "__spring_security_session_mgmt_filter_applied";
053        private SessionManagementFilter smTarget = null;
054        private ConcurrentSessionFilter csTarget = null;
055        private SessionRegistry sessionRegistry;
056        private LogoutHandler[] logoutHandlers = null;
057        private SecurityContextRepository securityContextRepository = null;
058        private SessionAuthenticationStrategy sessionStrategy = new SessionFixationProtectionStrategy();
059        private String invalidSessionUrl = null;
060    
061        @Override
062        public void setServletContext(ServletContext servletContext) {
063            this.servletContext = servletContext;
064        }
065    
066        private ConcurrentSessionFilter getCSTarget(ProceedingJoinPoint pjp) {
067            if (csTarget == null) {
068                csTarget = (ConcurrentSessionFilter) pjp.getTarget();
069            }
070            return csTarget;
071        }
072    
073        private SessionRegistry getSessionRegistry(ConcurrentSessionFilter target) throws
074                NoSuchFieldException, IllegalArgumentException, IllegalAccessException {
075            if (sessionRegistry == null) {
076                Field f = ConcurrentSessionFilter.class.getDeclaredField("sessionRegistry");
077                f.setAccessible(true);
078                sessionRegistry = (SessionRegistry) f.get(target);
079            }
080            return sessionRegistry;
081        }
082    
083        private LogoutHandler[] getLogoutHandlers(ConcurrentSessionFilter target) throws
084                NoSuchFieldException, IllegalArgumentException, IllegalAccessException {
085            if (logoutHandlers == null) {
086                Field f = ConcurrentSessionFilter.class.getDeclaredField("handlers");
087                f.setAccessible(false);
088                logoutHandlers = (LogoutHandler[]) f.get(target);
089            }
090            return logoutHandlers;
091        }
092    
093        private SessionManagementFilter getSMTarget(ProceedingJoinPoint pjp) {
094            if (smTarget == null) {
095                smTarget = (SessionManagementFilter) pjp.getTarget();
096            }
097            return smTarget;
098        }
099    
100        private SecurityContextRepository getSecurityContextRepository(SessionManagementFilter target) throws
101                NoSuchFieldException, IllegalArgumentException, IllegalAccessException {
102            if (securityContextRepository == null) {
103                Field f = SessionManagementFilter.class.getDeclaredField("securityContextRepository");
104                f.setAccessible(true);
105                securityContextRepository = (SecurityContextRepository) f.get(target);
106            }
107            return securityContextRepository;
108        }
109    
110        private String getInvalidSessionUrl(SessionManagementFilter target) throws
111                NoSuchFieldException, IllegalArgumentException, IllegalAccessException {
112            if (invalidSessionUrl == null) {
113                Field f = SessionManagementFilter.class.getDeclaredField("invalidSessionUrl");
114                f.setAccessible(true);
115                invalidSessionUrl = (String) f.get(target);
116            }
117            return invalidSessionUrl == null ? null : invalidSessionUrl.isEmpty() ? null : invalidSessionUrl;
118        }
119    
120        private FilterChain getFilterChain(JoinPoint jp) {
121            if (jp == null) {
122                return null;
123            } else {
124                for (Object obj : jp.getArgs()) {
125                    if (obj != null && obj instanceof FilterChain) {
126                        return (FilterChain) obj;
127                    }
128                }
129                return null;
130            }
131        }
132    
133        @Around("execution(* org.springframework.security.web.session.SessionManagementFilter.doFilter(..))")
134        public Object doSMFilter(ProceedingJoinPoint pjp) throws Throwable {
135            HttpHolder holder = HttpHolder.getInstance(pjp);
136            if (holder.isGwt()) {
137                if (logger.isDebugEnabled()) {
138                    logger.debug("GwtSessionManagementFilter instead of  SessionManagementFilter");
139                }
140                if (holder.getRequest().getAttribute(FILTER_APPLIED) != null) {
141                    if (logger.isDebugEnabled()) {
142                        logger.debug("Request already filter applied.");
143                    }
144                    //do next filter
145                    getFilterChain(pjp).doFilter(holder.getRequest(), holder.getResponse());
146                    return null;
147                }
148    
149                holder.getRequest().setAttribute(FILTER_APPLIED, Boolean.TRUE);
150    
151                if (!getSecurityContextRepository(getSMTarget(pjp)).containsContext(holder.getRequest())) {
152                    Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
153    
154                    if (authentication != null && !GwtResponseUtil.isAnonymous(authentication)) {
155                        // The user has been authenticated during the current request, so call the session strategy
156                        try {
157                            sessionStrategy.onAuthentication(authentication, holder.getRequest(), holder.getResponse());
158                        } catch (SessionAuthenticationException e) {
159                            if (logger.isErrorEnabled()) {
160                                logger.error("SessionAuthenticationStrategy rejected the authentication object", e);
161                            }
162                            SecurityContextHolder.clearContext();
163                            GwtResponseUtil.processGwtException(servletContext, holder.getRequest(), holder.getResponse(), e);
164                            return null;
165                        }
166                        // Eagerly save the security context to make it available for any possible re-entrant
167                        // requests which may occur before the current request completes. SEC-1396.
168                        getSecurityContextRepository(getSMTarget(pjp)).saveContext(SecurityContextHolder.getContext(), holder.getRequest(), holder.getResponse());
169                    } else {
170                        // No security context or authentication present. Check for a session timeout
171                        if (holder.getRequest().getRequestedSessionId() != null && !holder.getRequest().isRequestedSessionIdValid()) {
172                            if (logger.isDebugEnabled()) {
173                                logger.debug("Requested session ID" + holder.getRequest().getRequestedSessionId() + " is invalid.");
174                            }
175                            if (getInvalidSessionUrl(getSMTarget(pjp)) != null) {
176                                if (logger.isDebugEnabled()) {
177                                    logger.debug("Starting new session (if required) and notify front-end user");
178                                }
179                                holder.getRequest().getSession();
180                                GwtResponseUtil.processGwtException(servletContext, holder.getRequest(), holder.getResponse(),
181                                        new SessionAuthenticationException("Session had invalid."));
182                                return null;
183                            }
184                        }
185                    }
186                }
187    
188                getFilterChain(pjp).doFilter(holder.getRequest(), holder.getResponse());
189                return null;
190            } else {
191                //back to the original SessionManagementFilter
192                return pjp.proceed();
193            }
194        }
195    
196        @Around("execution(* org.springframework.security.web.session.ConcurrentSessionFilter.doFilter(..))")
197        public Object doCSFilter(ProceedingJoinPoint pjp) throws Throwable {
198            HttpHolder holder = HttpHolder.getInstance(pjp);
199            SessionInformation info = holder.getRequest().getSession(false) != null && GwtResponseUtil.isGwt(holder.getRequest())
200                    ? getSessionRegistry(getCSTarget(pjp)).getSessionInformation(holder.getRequest().getSession(false).getId()) : null;
201            if (info != null) {
202                if (info.isExpired()) {
203                    if (logger.isDebugEnabled()) {
204                        logger.debug("Concurrent Session ready logout");
205                    }
206                    Authentication auth = SecurityContextHolder.getContext().getAuthentication();
207                    for (LogoutHandler handler : getLogoutHandlers(getCSTarget(pjp))) {
208                        handler.logout(holder.getRequest(), holder.getResponse(), auth);
209                    }
210                    GwtResponseUtil.processGwtException(servletContext, holder.getRequest(), holder.getResponse(),
211                            new SessionAuthenticationException("This session has been expired (possibly due to multiple concurrent " +
212                                    "logins being attempted as the same user)."));
213                    return null;
214                } else {
215                    if (logger.isDebugEnabled()) {
216                        logger.debug("Concurrent info not expired");
217                    }
218                    // Non-expired - update last request date/time
219                    info.refreshLastRequest();
220                    getFilterChain(pjp).doFilter(holder.getRequest(), holder.getResponse());
221                    return null;
222                }
223            } else {
224                return pjp.proceed();
225            }
226        }
