040        private void validate(TreeLogger logger, String typeName) throws UnableToCompleteException {
041            if (!typeName.endsWith("Async")) {
042                logger.log(TreeLogger.ERROR, format("Asynchronous service's name must ends with \"Async\",Obviously \"%s\" doesn't", typeName));
043                throw new UnableToCompleteException();
044            }
045            TypeOracle typeOracle = this.context.getTypeOracle();
046            this.typeName = typeName;
047            try {
048                this.sourceType = typeOracle.getType(typeName);
049                if (this.sourceType.isInterface() == null) {
050                    logger.log(TreeLogger.ERROR, format("%s is not an interface.", sourceType.getParameterizedQualifiedSourceName()), null);
051                    throw new UnableToCompleteException();
052                }
053                JClassType loginableAsyncType = typeOracle.getType(LoginableAsync.class.getName());
054                if (!sourceType.getFlattenedSupertypeHierarchy().contains(loginableAsyncType)) {
055                    logger.log(TreeLogger.ERROR, format("Type: %s not extends LoginableAsync", typeName));
056                    throw new UnableToCompleteException();
057                }
058                this.packageName = this.sourceType.getPackage().getName();
059            } catch (NotFoundException ex) {
060                logger.log(TreeLogger.ERROR, format("validate %s fail", typeName), ex);
061                throw new UnableToCompleteException();
062            }
063            this.serviceName = typeName.substring(0, this.typeName.length() - 5);
064            this.serviceType = typeOracle.findType(serviceName);
065            if (serviceType == null) {
066                logger.log(TreeLogger.ERROR, format("Could not find remote service type : %s", serviceName));
067                throw new UnableToCompleteException();
068            }
069            if (this.serviceType.isInterface() == null) {
070                logger.log(TreeLogger.ERROR, format("%s is not an interface.", serviceType.getParameterizedQualifiedSourceName()), null);
071                throw new UnableToCompleteException();
072            }
073            //Find Remote Service
074            JClassType remoteServiceType = typeOracle.findType(RemoteService.class.getName());
075            if (!serviceType.getFlattenedSupertypeHierarchy().contains(remoteServiceType)) {
076                logger.log(TreeLogger.ERROR, format("Type: %s not extends RemoteService", serviceName));
077                throw new UnableToCompleteException();
078            }
079        }

081        @Override
082        public String generate(TreeLogger logger, GeneratorContext context, String typeName) throws UnableToCompleteException {
083            this.context = context;
084            validate(logger, typeName);
085            this.className = format("%s_Proxy", this.sourceType.getSimpleSourceName());
086            generateClass(logger);
087            return this.packageName + "." + this.className;
088        }
089    
090        private void generateClass(TreeLogger logger) {
091            PrintWriter printWriter = context.tryCreate(logger, this.packageName, this.className);
092            if (printWriter == null) {
093                return;
094            }
095            ClassSourceFileComposerFactory composer = new ClassSourceFileComposerFactory(this.packageName, this.className);
096            composer.addImport("com.google.gwt.core.client.GWT");
097            composer.addImport("com.google.gwt.core.client.Scheduler");
098            composer.addImport("com.google.gwt.event.shared.HandlerRegistration");
099            composer.addImport("com.google.gwt.user.client.rpc.AsyncCallback");
100            composer.addImport("com.gwt.ss.client.GwtAccessDeniedException");
101            composer.addImport("com.gwt.ss.client.GwtSecurityException");
102            composer.addImport("com.gwt.ss.client.loginable.HasLoginHandler");
103            composer.addImport("com.gwt.ss.client.loginable.LoginEvent");
104            composer.addImport("com.gwt.ss.client.loginable.LoginHandler");
105            composer.addImport("com.gwt.ss.client.loginable.LoginableService");
106            composer.addImport("com.gwt.ss.client.loginable.LoginCancelException");
107            composer.addImplementedInterface(this.typeName);
108            composer.addImplementedInterface(format("%s<%s>", LoginableService.class.getName(), this.typeName));
109            SourceWriter writer = composer.createSourceWriter(context, printWriter);
110            generateFields(writer);
111            generateMethods(writer);
112            writer.outdent();
113            writer.println("}");
114            context.commit(logger, printWriter);
115        }

117        private void generateFields(SourceWriter writer) {
118            String serviceTypeName = this.typeName.substring(0, this.typeName.length() - 5);
119            writer.println("private HasLoginHandler hasLoginHandler = null;");
120            writer.println("private %s service = null;", this.typeName);
121            writer.println();
122            writer.println("@Override");
123            writer.println("public HasLoginHandler getHasLoginHandler(){");
124            writer.indentln("return hasLoginHandler;");
125            writer.println("}");
126            writer.println();
127            writer.println("@Override");
128            writer.println("public void setHasLoginHandler(HasLoginHandler hasLoginHandler){");
129            writer.indentln("this.hasLoginHandler = hasLoginHandler;");
130            writer.println("}");
131            writer.println();
132            writer.println("@Override");
133            writer.println("public %s getRemoteService(){", this.typeName);
134            writer.indent();
135            writer.println("if(this.service ==null){");
136            writer.indentln("this.service = (%s) GWT.create(%s.class);", this.typeName, serviceTypeName);
137            writer.println("}");
138            writer.println("return this.service;");
139            writer.outdent();
140            writer.println("}");
141            writer.println();
142            writer.println("@Override");
143            writer.println("public void setRemoteService(%s service){", this.typeName);
144            writer.indentln("this.service = service;");
145            writer.println("}");
146            writer.println();
147        }

149        private void generateMethods(SourceWriter writer) {
150            for (JMethod method : this.serviceType.getMethods()) {
151                generateMethod(writer, method);
152            }
153    
154            for (JClassType parentType : this.serviceType.getImplementedInterfaces()) {
155                for (JMethod method : parentType.getMethods()) {
156                    generateMethod(writer, method);
157                }
158            }
159        }

161        private void generateMethod(SourceWriter writer, JMethod method) {
162            writer.println("@Override");
163            writer.print("public void %s(", method.getName());
164            boolean first = true;
165            for (JParameter param : method.getParameters()) {
166                if (first) {
167                    first = false;
168                } else {
169                    writer.print(", ");
170                }
171                writer.print("final %s %s", param.getType().getParameterizedQualifiedSourceName(), param.getName());
172            }
173            writer.println(", final AsyncCallback<%s> callback) {", method.getReturnType().getParameterizedQualifiedSourceName());
174            writer.indent();
175            writer.print("getRemoteService().%s(", method.getName());
176            first = true;
177            for (JParameter param : method.getParameters()) {
178                if (first) {
179                    first = false;
180                } else {
181                    writer.print(", ");
182                }
183                writer.print(param.getName());
184            }
185            writer.println(", new AsyncCallback<%s>() {", method.getReturnType().getParameterizedQualifiedSourceName());
186            writer.println();
187            writer.indent();
188            writer.println("@Override");
189            writer.println("public void onSuccess(%s result) {", method.getReturnType().getParameterizedQualifiedSourceName());
190            writer.indentln("callback.onSuccess(result);");
191            writer.println("}");
192            writer.println();
193            writer.println("@Override");
194            writer.println("public void onFailure(Throwable caught) {");
195            writer.indent();
196            writer.println("if(caught instanceof GwtAccessDeniedException || !(caught instanceof GwtSecurityException) || getHasLoginHandler()==null){");
197            writer.indentln("callback.onFailure(caught);");
198            writer.println("} else {");
199            writer.indent();
200            writer.println("LoginHandler lh = new LoginHandler() {");
201            writer.indent();
202            writer.println();
203            writer.println("private HandlerRegistration hr;");
204            writer.println();
205            writer.println("@Override");
206            writer.println("public void setLoginHandlerRegistration(HandlerRegistration hr) {");
207            writer.indentln("this.hr = hr;");
208            writer.println("}");
209            writer.println();
210            writer.println("@Override");
211            writer.println("public void onLogin(LoginEvent e) {");
212            writer.indent();
213            writer.println("if (hr != null) {");
214            writer.indentln("hr.removeHandler();");
215            writer.println("}");
216            writer.println("GWT.log(\"Receive login \"+(e.isCanceled()?\"cancel\":\"succeed\")+\" event!\");");
217            writer.println("if(e.isCanceled()){");
218            writer.indentln("callback.onFailure(new LoginCancelException(\"user cancel login process.\"));");
219            writer.println("} else {");
220            writer.indent();
221            writer.println("Scheduler.get().scheduleDeferred(new Scheduler.ScheduledCommand() {");
222            writer.println();
223            writer.indent();
224            writer.println("@Override");
225            writer.println("public void execute() {");
226            writer.indent();
227            writer.print("%s(", method.getName());
228            first = true;
229            for (JParameter param : method.getParameters()) {
230                if (first) {
231                    first = false;
232                } else {
233                    writer.print(", ");
234                }
235                writer.print(param.getName());
236            }
237            writer.println(", callback);");
238            writer.outdent();
239            writer.println("}");
240            writer.outdent();
241            writer.println("});");
242            writer.outdent();
243            writer.println("}");
244            writer.outdent();
245            writer.println("}");
246            writer.outdent();
247            writer.println("};");
248            writer.println("lh.setLoginHandlerRegistration(getHasLoginHandler().addLoginHandler(lh));");
249            writer.println("getHasLoginHandler().startLogin(caught);");
250            writer.outdent();
251            writer.println("}");
252            writer.outdent();
253            writer.println("}");
254            writer.outdent();
255            writer.println("});");
256            writer.outdent();
257            writer.println("}");
258            writer.println();
259        }
