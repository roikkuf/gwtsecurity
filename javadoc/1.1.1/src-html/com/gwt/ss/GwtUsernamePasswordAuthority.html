047     * When &lt;form-login&gt; come from GWT RPC, return void message when success and prone

049     * effect with web page form-login.<br/>
050     * Note: Certification extracting code provide by Amit Khanna<br/>

108         * Provide by Amit Khanna<br/>
109         * 由Amit Khanna提供
110         */
111        private PayloadInfo extract(JoinPoint jp) throws IOException, ServletException {
112            PayloadInfo result = payloadHolder.get();
113            HttpHolder httpHolder = HttpHolder.getInstance(jp);
114            HttpServletRequest request = httpHolder.getRequest();
115            String username = null;
116            String password = null;
117            boolean rememberMe = false;
118            boolean forceLogout = false;
119            if (result == null && request != null) {
120                String payload = RPCServletUtils.readContentAsGwtRpc(request);
121                RPCRequest rpcRequest = RPC.decodeRequest(payload, null, serializationPolicyProvider);
122                Object[] requestParams = rpcRequest.getParameters();
123                assert requestParams.length > 1 : "parameter count incorrect";
124                username = (String) requestParams[0];
125                password = (String) requestParams[1];
126                if (requestParams.length > 2) {
127                    try {
128                        rememberMe = (Boolean) requestParams[2];
129                    } catch (Exception e) {
130                    }
131                }
132                if (requestParams.length > 3) {
133                    try {
134                        forceLogout = (Boolean) requestParams[3];
135                    } catch (Exception e) {
136                    }
137                }
138                if (username != null && password != null) {
139                    result = new PayloadInfo(username, password, httpHolder, rememberMe, forceLogout);
140                    payloadHolder.set(result);
141                }
142            }
143            return result;
144        }

146        @SuppressWarnings({ "rawtypes", "unchecked" })
147        @Around("execution(* org.springframework.security.web.authentication.AbstractAuthenticationProcessingFilter.doFilter(..))")
148        public Object doFilter(ProceedingJoinPoint pjp) throws Throwable {
149            HttpHolder httpHolder = HttpHolder.getInstance(pjp);
150            AbstractAuthenticationProcessingFilter filter = (AbstractAuthenticationProcessingFilter) pjp.getTarget();
151            if (httpHolder.isGwt() && requiresAuthentication(filter, httpHolder.getRequest())) {
152                PayloadInfo pi = null;
153                try {
154                    pi = extract(pjp);
155                    UsernamePasswordAuthenticationToken authRequest = new UsernamePasswordAuthenticationToken(pi.getUsername(), pi.getPassword());
156                    Class<AbstractAuthenticationProcessingFilter> c = AbstractAuthenticationProcessingFilter.class;
157                    Field f = c.getDeclaredField("allowSessionCreation");
158                    f.setAccessible(true);
159                    boolean allowSessionCreation = (Boolean) f.get(filter);
160                    HttpSession session = httpHolder.getRequest().getSession(allowSessionCreation);
161                    if (session != null) {
162                        httpHolder.getRequest().getSession().setAttribute(GwtResponseUtil.SPRING_SECURITY_LAST_USERNAME_KEY,
163                                TextEscapeUtils.escapeEntities(pi.getUsername()));
164                    }
165                    f = c.getDeclaredField("authenticationDetailsSource");
166                    f.setAccessible(true);
167                    authRequest.setDetails(((AuthenticationDetailsSource) f.get(filter)).buildDetails(httpHolder.getRequest()));
168                    Authentication authResult = getAuthenticationManager().authenticate(authRequest);
169                    SecurityContextHolder.getContext().setAuthentication(authResult);
170                    filter.getRememberMeServices().loginSuccess(pi.isRememberMe()
171                            ? new RemeberRequestWrapper(httpHolder.getRequest(), rememberMeParameter)
172                            : httpHolder.getRequest(), httpHolder.getResponse(), authResult);
173    
174                    //Patch provided by Steven Jardine steven.j...@gmail.com http://code.google.com/u/@UhVVQ1JZAxVEXgF4/
175                    f = c.getDeclaredField("sessionStrategy");
176                    f.setAccessible(true);
177                    SessionAuthenticationStrategy sessionStrategy = (SessionAuthenticationStrategy) f.get(filter);
178                    try {
179                        sessionStrategy.onAuthentication(authResult, httpHolder.getRequest(), httpHolder.getResponse());
180                    } catch (SessionAuthenticationException e) {
181                        if (pi.isForceLogout()) {
182                            SessionRegistry registry = applicationContext.getBean(SessionRegistry.class);
183                            if (registry != null) {
184                                for (SessionInformation info : registry.getAllSessions(authResult.getPrincipal(), true)) {
185                                    info.expireNow();
186                                }
187                            }
188                            sessionStrategy.onAuthentication(authResult, httpHolder.getRequest(), httpHolder.getResponse());
189                        } else {
190                            throw e;
191                        }
192                    }
193                    applicationContext.publishEvent(new InteractiveAuthenticationSuccessEvent(authResult, this.getClass()));
194                    return null;
195                } catch (Exception e) {
196                    if (logger.isErrorEnabled() && !isSuppressLoginErrorMessages()) {
197                        logger.error("Gwt login fail:", e);
198                    }
199                    GwtResponseUtil.processGwtException(servletContext, httpHolder.getRequest(), httpHolder.getResponse(), e);
200                    return null;
201                } finally {
202                    payloadHolder.remove();
203                }
204            } else {
205                return pjp.proceed();
206            }
207        }

209        protected boolean requiresAuthentication(AbstractAuthenticationProcessingFilter filter, HttpServletRequest request) {
210            String uri = request.getRequestURI();
211            if (logger.isDebugEnabled()) {
212                logger.debug("RequiresAuthentication check for \n"
213                        + "url = " + uri + "\ncontext path = " + request.getContextPath() + "\nprocessUrl = " + filter.getFilterProcessesUrl());
214            }
215            if (uri == null || uri.isEmpty()) {
216                return false;
217            } else {
218                int pathParamIndex = uri.indexOf(';');
219    
220                if (pathParamIndex > 0) {
221                    // strip everything after the first semi-colon
222                    uri = uri.substring(0, pathParamIndex);
223                }
224    
225                if ("".equals(request.getContextPath())) {
226                    return uri.endsWith(filter.getFilterProcessesUrl());
227                }
228                return uri.endsWith(request.getContextPath() + filter.getFilterProcessesUrl());
229            }
230        }

232        @Override
233        public void onApplicationEvent(InteractiveAuthenticationSuccessEvent event) {
234            PayloadInfo pi = payloadHolder.get();
235            if (pi != null && pi.getHttpHolder().isGwt()) {
236                HttpServletRequest request = pi.getHttpHolder().getRequest();
237                Authentication authResult = event.getAuthentication();
238                if (logger.isDebugEnabled()) {
239                    logger.debug("Gwt authentication success. Updating SecurityContextHolder to contain: " + authResult);
240                }
241                GwtResponseUtil.writeResponse(servletContext, request, pi.getHttpHolder().getResponse(),
242                        String.format("//OK[[],%s,%s]", AbstractSerializationStream.DEFAULT_FLAGS,
243                        AbstractSerializationStream.SERIALIZATION_STREAM_VERSION));
244            }
245        }

302        private class RemeberRequestWrapper extends HttpServletRequestWrapper {
303    
304            private String rememberMeParameter = "_spring_security_remember_me";
305    
306            public RemeberRequestWrapper(HttpServletRequest request, String rememberMeParameter) {
307                super(request);
308                if (rememberMeParameter != null && !rememberMeParameter.isEmpty()) {
309                    this.rememberMeParameter = rememberMeParameter;
310                }
311            }
312    
313            @Override
314            public String getParameter(String name) {
315                return name.equals(rememberMeParameter) ? "true" : super.getParameter(name);
316            }
317        }
